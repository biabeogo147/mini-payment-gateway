\apiblockheading{Business rules and usage notes.}
\begin{itemize}
  \item These routes are read-oriented and do not change business state.

  suspended or disabled.

\apiblockheading{Endpoints.}
\begin{longtable}{@{}L{0.35\textwidth}L{0.55\textwidth}@{}}
\toprule
\textbf{Endpoint group} & \textbf{Purpose} \\
\midrule
\endfirsthead
\toprule
\textbf{Endpoint group} & \textbf{Purpose} \\
\midrule
\endhead
\code{POST /v1/ops/merchants} & Register a new merchant in pending-review
state. \\
\bodyrowrule
\routecell{\code{PUT /v1/ops/merchants/{merchant_id}/onboarding-case}\\
\code{POST /v1/ops/merchants/{merchant_id}/onboarding-case/approve}\\
\code{POST /v1/ops/merchants/{merchant_id}/onboarding-case/reject}} & Submit,
approve, or reject the merchant's onboarding case. \\
\bodyrowrule
\routecell{\code{POST /v1/ops/merchants/{merchant_id}/activate}\\
\code{POST /v1/ops/merchants/{merchant_id}/suspend}\\
\code{POST /v1/ops/merchants/{merchant_id}/disable}} & Move a merchant through
operational lifecycle states. \\
\bodyrowrule
\routecell{\code{POST /v1/ops/merchants/{merchant_id}/credentials}\\
\code{POST /v1/ops/merchants/{merchant_id}/credentials/rotate}} & Create the
first active credential or rotate the active credential set. \\
\bottomrule
\end{longtable}

\apiblockheading{Business rules and usage notes.}
\begin{itemize}
  \item Merchant creation starts in \code{PENDING_REVIEW}.
  \item Activation requires an approved onboarding case and an active
  credential.
  \item Credential rotation and merchant disable are restricted to
  administrator-level access.
  \item Operational write actions expect a human-readable reason so the action
  can be audited later.
  \item Raw credential secrets are accepted during create or rotate actions but
  are not returned in metadata reads afterward.
\end{itemize}
