\section{Restricted And Internal APIs}

\restrictedbox{These endpoints are for internal operators, support staff, and
operational recovery workflows. Merchant teams usually encounter them only
during onboarding, account provisioning, or support-led troubleshooting. They
are not intended for direct merchant integration.}

\toprule
\textbf{Endpoint} & \textbf{Purpose} \\
\midrule
\endfirsthead
\toprule
\textbf{Endpoint} & \textbf{Purpose} \\
\midrule
\endhead
\code{GET /v1/internal/auth/bootstrap-status} & Show whether the first

\bodyrowrule
\code{POST /v1/internal/auth/bootstrap} & Create the first internal
administrator when bootstrap is still open. \\
\bodyrowrule
\routecell{\code{POST /v1/internal/auth/login}\\
\code{POST /v1/internal/auth/logout}\\
\code{GET /v1/internal/auth/me}\\
\code{POST /v1/internal/auth/change-password}} & Standard session lifecycle for
internal operators after bootstrap is closed. \\
\bottomrule

\apiblockheading{Business rules and usage notes.}
\begin{itemize}
  \item Bootstrap is a one-time setup action for the first internal
  administrator.
  \item After bootstrap, all normal internal access uses session-based sign-in.
  \item Permission checks are enforced on the server side according to the
  signed-in operator role.
  \item Password changes refresh the operator's own session and invalidate older
  credentials.
\end{itemize}

\subsection{Internal User Administration}

\begin{longtable}{@{}L{0.25\textwidth}L{0.65\textwidth}@{}}
\caption{Internal user administration family summary}\\
\toprule
\textbf{Item} & \textbf{Details} \\
\midrule
\endfirsthead
\toprule
\textbf{Item} & \textbf{Details} \\
\midrule
\endhead
Purpose & Create, update, and reset internal operator accounts. \\
\bodyrowrule
Intended audience & Internal administrators only. \\
\bodyrowrule
Authentication & Internal session cookie with administrator role. \\
\bodyrowrule
Reference IDs & \apiid{OPS-06}. \\
\bottomrule
\end{longtable}

\apiblockheading{Endpoints.}
\begin{longtable}{@{}L{0.34\textwidth}L{0.56\textwidth}@{}}
\toprule
\textbf{Endpoint} & \textbf{Purpose} \\
\midrule
\endfirsthead
\toprule
\textbf{Endpoint} & \textbf{Purpose} \\
\midrule
\endhead
\code{GET /v1/internal/users} & List internal operator accounts. \\
\bodyrowrule
\code{POST /v1/internal/users} & Create a new internal operator account. \\
\bodyrowrule
\code{PATCH /v1/internal/users/{user_id}} & Update the role, status, or profile
of an internal operator account. \\
\bodyrowrule
\code{POST /v1/internal/users/{user_id}/reset-password} & Generate and return a
new one-time password for an internal operator account. \\
\bottomrule
\end{longtable}

\apiblockheading{Business rules and usage notes.}
\begin{itemize}
  \item This family is administrator-only.
  \item Generated passwords are returned once on create or reset and are not
  retrievable later.
  \item The last active administrator cannot be deactivated by mistake.
\end{itemize}
