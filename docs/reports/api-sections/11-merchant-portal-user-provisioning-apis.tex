\code{GET /v1/ops/merchants/{merchant_id}/portal-users} & List dashboard user

\bodyrowrule
\code{POST /v1/ops/merchants/{merchant_id}/portal-users} & Create a dashboard

\bodyrowrule
\code{PATCH /v1/ops/merchants/{merchant_id}/portal-users/{user_id}} & Update a
dashboard user's profile, role, or status. \\
\bodyrowrule
\code{POST /v1/ops/merchants/{merchant_id}/portal-users/{user_id}/reset-password}
& Generate and return a new one-time password for a dashboard user. \\
\bottomrule
\end{longtable}

\apiblockheading{Business rules and usage notes.}
\begin{itemize}
  \item This family is administrator-only.
  \item Generated passwords are shown once on create or reset and are not
  retrievable later.
  \item Standard operations users can continue merchant support workflows, but
  they cannot provision or manage Merchant Dashboard user accounts.
\end{itemize}
